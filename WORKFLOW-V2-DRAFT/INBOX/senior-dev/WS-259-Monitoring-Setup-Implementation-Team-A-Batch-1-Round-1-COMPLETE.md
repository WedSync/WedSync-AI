// Wedding day detection (affects P0 escalation decisions)
const isWeddingDay = (): boolean => {
  const today = new Date();
  const dayOfWeek = today.getDay();
  return dayOfWeek === 0 || dayOfWeek === 6; // Sunday or Saturday
};
